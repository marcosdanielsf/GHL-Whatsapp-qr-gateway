### Logica do Router

```typescript
// src/services/router.ts

interface SendRequest {
  type: 'text' | 'media' | 'call' | 'button' | 'list';
  to: string;
  content: any;
}

async function route(request: SendRequest): Promise<Provider> {
  // Chamadas SEMPRE vao pro Z-API
  if (request.type === 'call') {
    return 'zapi';
  }

  // Se Baileys estiver conectado, usa Baileys
  if (await baileys.isConnected()) {
    return 'baileys';
  }

  // Fallback para Z-API se Baileys cair
  if (config.zapiEnabled) {
    return 'zapi';
  }

  throw new Error('Nenhum provider disponivel');
}
```

## IMPLEMENTACAO DO SERVICO Z-API

### src/services/zapi.service.ts

```typescript
import axios from 'axios';

interface ZApiConfig {
  instanceId: string;
  token: string;
  clientToken: string;
}

interface CallRequest {
  phone: string;
  callDuration?: number;
}

interface CallResponse {
  zaapId: string;
  messageId: string;
  id: string;
}

class ZApiService {
  private baseUrl = 'https://api.z-api.io';

  constructor(private config: ZApiConfig) {}

  private get headers() {
    return {
      'Content-Type': 'application/json',
      'Client-Token': this.config.clientToken
    };
  }

  private get endpoint() {
    return `${this.baseUrl}/instances/${this.config.instanceId}/token/${this.config.token}`;
  }

  // ========== CHAMADAS ==========

  async makeCall(request: CallRequest): Promise<CallResponse> {
    const response = await axios.post(
      `${this.endpoint}/send-call`,
      {
        phone: this.formatPhone(request.phone),
        callDuration: request.callDuration || 10
      },
      { headers: this.headers }
    );

    return response.data;
  }

  async rejectCall(callId: string, callCreator: string): Promise<void> {
    await axios.post(
      `${this.endpoint}/reject-call`,
      { callId, callCreator },
      { headers: this.headers }
    );
  }

  // ========== FALLBACK MENSAGENS ==========

  async sendText(phone: string, text: string): Promise<any> {
    const response = await axios.post(
      `${this.endpoint}/send-text`,
      {
        phone: this.formatPhone(phone),
        message: text
      },
      { headers: this.headers }
    );

    return response.data;
  }

  async sendMedia(phone: string, type: string, url: string, caption?: string): Promise<any> {
    const response = await axios.post(
      `${this.endpoint}/send-${type}`,
      {
        phone: this.formatPhone(phone),
        [type]: url,
        caption
      },
      { headers: this.headers }
    );

    return response.data;
  }

  // ========== STATUS ==========

  async getStatus(): Promise<{ connected: boolean; phone?: string }> {
    try {
      const response = await axios.get(
        `${this.endpoint}/status`,
        { headers: this.headers }
      );
      return {
        connected: response.data.connected,
        phone: response.data.phone
      };
    } catch {
      return { connected: false };
    }
  }

  // ========== UTILS ==========

  private formatPhone(phone: string): string {
    // Remove tudo que nao e numero
    let cleaned = phone.replace(/\D/g, '');

    // Adiciona 55 se nao tiver
    if (!cleaned.startsWith('55')) {
      cleaned = '55' + cleaned;
    }

    return cleaned;
  }
}

export { ZApiService, ZApiConfig, CallRequest, CallResponse };
```

## WEBHOOKS Z-API (Receber eventos)

### Configurar no Z-API Dashboard

```
URL: https://seu-gateway.railway.app/api/zapi/webhook
Eventos: Todos
```

### src/routes/zapi.webhook.ts

```typescript
import { Router } from 'express';
import { supabase } from '../lib/supabase';

const router = Router();

router.post('/webhook', async (req, res) => {
  const { event, data } = req.body;

  switch (event) {
    case 'call':
      await handleCallEvent(data);
      break;

    case 'message':
      // Fallback - se Z-API receber msg, repassa pro fluxo normal
      await handleMessageFallback(data);
      break;

    case 'status':
      await handleStatusEvent(data);
      break;
  }

  res.status(200).send('OK');
});

async function handleCallEvent(data: any) {
  const { callId, status, phone, isVideo, duration } = data;

  // Registra no banco
  await supabase.from('ghl_wa_call_logs').insert({
    direction: 'inbound',
    phone,
    status,
    provider: 'zapi',
    zapi_call_id: callId
  });

  // Se configurado, notifica GHL
  if (process.env.GHL_CALL_WEBHOOK_URL) {
    await axios.post(process.env.GHL_CALL_WEBHOOK_URL, {
      type: 'call_received',
      phone,
      status,
      isVideo
    });
  }
}

export default router;
```
